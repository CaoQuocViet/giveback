1. NguoiDung (Abstract class)
- Thuoc tinh:
  + id: string
  + hoTen: string 
  + email: string
  + soDienThoai: string
  + matKhau: string
  + vaiTro: VaiTro
  + anhDaiDien: string
  + tinh: string
  + quanHuyen: string
  + xaPhuong: string
  + diaChiChiTiet: string
  + thoiGianTao: Date
  + thoiGianCapNhat: Date

2. QuanTriVien extends NguoiDung
- Thuoc tinh:
  + laQuanTriHeThong: boolean
- Phuong thuc:
  + xacThucToChuc(idToChuc: string): Promise<boolean>
  + xemDanhSachToChuc(): Promise<ToChucTuThien[]>
  + giamSatChienDich(): Promise<ChienDich[]>

3. NguoiDongGop extends NguoiDung
- Phuong thuc:
  + dongGop(idChienDich: string, soTien: decimal): Promise<boolean>
  + xemLichSuDongGop(): Promise<DongGop[]>
  + danhGiaChienDich(idChienDich: string, diem: number): Promise<boolean>

4. NguoiNhanHoTro extends NguoiDung
- Thuoc tinh:
  + trangThaiXacThuc: boolean
- Phuong thuc:
  + xemDanhSachNhanHoTro(): Promise<HoTro[]>
  + guiPhanHoi(idChienDich: string, noiDung: string): Promise<boolean>

5. ToChucTuThien extends NguoiDung
- Thuoc tinh:
  + giayPhepUrl: string
  + anhXacMinh: string[]
  + trangThaiXacThuc: TrangThaiXacThuc
  + diemDanhGia: decimal
  + soChienDich: number
  + soChienDichHoanThanh: number
  + tongSoTienHuyDong: decimal
  + tongSoTienDaPhanPhoi: decimal
  + soNguoiNhanHoTro: number
  + thoiGianCapNhatThongKe: Date
- Phuong thuc:
  + taoChienDich(duLieu: DuLieuChienDich): Promise<ChienDich>
  + quanLyChienDich(idChienDich: string): Promise<boolean>
  + phanPhoiHoTro(duLieuHoTro: DuLieuHoTro): Promise<boolean>
  + taoBaoCao(idChienDich: string): Promise<BaoCao>
  + xoaChienDich(idChienDich: string): Promise<boolean>
  + suaChienDich(idChienDich: string, duLieu: DuLieuChienDich): Promise<boolean>
  + taoDongGopTrungGian(duLieuDongGop: DuLieuDongGop): Promise<boolean>

  + moTa: string
  + trangThai: TrangThaiChienDich

  + tinh: string
  + quanHuyen: string
  + xaPhuong: string

- Phuong thuc:

  + phanPhoiHoTro(hoTro: HoTro): Promise<boolean>
  + taoBaoCao(): Promise<BaoCao>

7. DongGop

- Thuoc tinh:
