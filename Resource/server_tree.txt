app/ - Frontend Application: Chứa các pages và components của Next.js application

components/ - Shared Components: Chứa các UI components có thể tái sử dụng

../givaback-server/ - Backend Services: code backend viết ở đây

src/
├── config/                 # Cấu hình hệ thống
│   ├── postgresConfig.js   # Cấu hình database
│   └── constants.js        # Các hằng số của hệ thống
│
├── controllers/            # Xử lý logic nghiệp vụ
│   ├── authController.js   # Xử lý đăng nhập/đăng ký
│   ├── campaignController.js    # Quản lý chiến dịch
│   ├── donationController.js    # Xử lý đóng góp
│   ├── distributionController.js # Quản lý phân phối hỗ trợ
│   ├── reportController.js      # Quản lý báo cáo
│   └── userController.js        # Quản lý người dùng
│
├── middleware/            # Middleware
│   ├── auth.js           # Xác thực JWT
│   ├── upload.js         # Xử lý upload file
│   ├── validate.js       # Validate dữ liệu
│   └── roleCheck.js      # Kiểm tra vai trò người dùng
│
├── models/               # Schema và models
│   ├── Campaign.js       # Model chiến dịch
│   ├── Donation.js       # Model đóng góp
│   ├── Distribution.js   # Model phân phối
│   ├── Report.js         # Model báo cáo
│   └── User.js           # Model người dùng
│
├── routes/               # Định tuyến API
│   ├── authRoutes.js     # Route xác thực
│   ├── campaignRoutes.js # Route chiến dịch
│   ├── donationRoutes.js # Route đóng góp
│   ├── reportRoutes.js   # Route báo cáo
│   └── userRoutes.js     # Route người dùng
│
├── services/            # Business logic
│   ├── paymentService.js    # Xử lý thanh toán
│   ├── notificationService.js # Xử lý thông báo
│   ├── uploadService.js      # Xử lý upload
│   └── reportService.js      # Xử lý báo cáo
│
├── utils/              # Tiện ích
│   ├── logger.js       # Ghi log
│   ├── validation.js   # Hàm validate
│   └── helpers.js      # Các hàm tiện ích
|
├── migrations/            # Quản lý thay đổi DB
│   ├── YYYYMMDD-create-users.js
│   └── YYYYMMDD-add-column.js
│
└── server.js          # Entry point
